# 1. Sistema e Runtime (A base do executável)

gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -c system/crt0.c -o system/crt0.o
gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -I. -c system/string.c -o system/string.o
gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -I. -c system/malloc.c -o system/malloc.o

# 2. Biblioteca Visual e RAD (Note que adicionei -fno-pie aqui também)

gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -Isystem -I. -c Runtime/controls.c -o Runtime/controls.o

gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -Isystem -I. -c Runtime/temp.c -o Runtime/temp.o

COMPONENTES 

gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -Isystem -I. -c Runtime/components/os_label.c -o Runtime/components/os_label.o

gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -Isystem -I. -c Runtime/components/os_button.c -o Runtime/components/os_button.o

gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -Isystem -I. -c Runtime/components/os_edit.c -o Runtime/components/os_edit.o

gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -Isystem -I. -c Runtime/components/os_image.c -o Runtime/components/os_image.o

gcc -m64 -ffreestanding -fno-stack-protector -fno-pie -Isystem -I. -c Runtime/components/os_radiobutton.c -o Runtime/components/os_radiobutton.o

# Usamos o LD (Linker) diretamente para ter controle total

ld -m elf_x86_64 -T user.ld -static -o tamp.elf \
    system/crt0.o \
    Runtime/temp.o \
    Runtime/controls.o \
    Runtime/components/os_form.o \
    Runtime/components/os_button.o \
    Runtime/components/os_edit.o \
    Runtime/components/os_image.o \
    Runtime/components/os_label.o \
    Runtime/components/os_checkbox.o \
    Runtime/components/os_panel.o \
    Runtime/components/os_memo.o \
    Runtime/components/os_combobox.o \
    Runtime/components/os_radiobutton.o \
    gui/gui.o \
    gui/wm.o \
    gui/z_order.o \
    gui/gui_interaction.o \
    system/graphics.o \
    system/fonts.o \
    system/string.o \
    events/input_events.o \
    Runtime/components/TOS_IPC.o \
    Runtime/components/TOSSerial.o \
    system/malloc.o

mcopy -o -i disco_sata.img tamp.elf ::/
